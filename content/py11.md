```python
# Program 28: Largest in List
def main():
    nums = list(map(int, input("Enter numbers: ").split()))
    
    largest = nums[0]
    
    for n in nums:
        if n > largest:
            largest = n
    
    print("Largest:", largest)

main()
```

    Enter numbers:  6 7 8
    

    Largest: 8
