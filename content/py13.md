# Program 33: Count Odd
def main():
    nums = list(map(int, input("Enter numbers: ").split()))
    
    count = 0
    
    for n in nums:
        if n % 2 != 0:
            count += 1
    
    print("Odd count:", count)

main()
```

    Enter numbers:  1 23 2 4 5 6 
    

    Odd count: 3

# Program 34: Reverse List
def main():
    nums = list(map(int, input("Enter numbers: ").split()))
    
    rev = nums[::-1]
    
    print("Reversed:", rev)

main()
```

    Reversed: [7, 6, 4, 3, 1]
    


```python
# Program 35: Remove Duplicates
def main():
    nums = list(map(int, input("Enter numbers: ").split()))
    
    unique = []
    
    for n in nums:
        if n not in unique:
            unique.append(n)
    
    print("Unique list:", unique)

main()
```

    Enter numbers:  3 4 5 6 7 8 
    

    Unique list: [3, 4, 5, 6, 7, 8]
