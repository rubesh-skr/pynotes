```python
# Program 37: Second Largest
def main():
    nums = list(map(int, input("Enter numbers: ").split()))
    
    nums = list(set(nums))
    nums.sort()
    
    print("Second largest:", nums[-2])

main()
```

    Enter numbers:  6 78 9 8 5 4 
    

    Second largest: 9
    


```python
# Program 38: Merge Lists
def main():
    a = list(map(int, input("Enter list1: ").split()))
    b = list(map(int, input("Enter list2: ").split()))
    
    merged = a + b
    
    print("Merged list:", merged)

main()
```
