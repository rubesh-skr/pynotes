```python
# Program 75: Find Index
def main():
    nums = list(map(int, input("Enter numbers: ").split()))
    target = int(input("Enter element: "))
    
    for i in range(len(nums)):
        if nums[i] == target:
            print("Index:", i)
            break

main()
```

```python
# Program 76: Count Occurrence
def main():
    nums = list(map(int, input("Enter numbers: ").split()))
    target = int(input("Enter element: "))
    
    count = 0
    
    for n in nums:
        if n == target:
            count += 1
    
    print("Count:", count)

main()
```

    Enter numbers:  1 2 3 4 5
    Enter element:  12345

```python
# Program 77: Replace Element
def main():
    nums = list(map(int, input("Enter numbers: ").split()))
    old = int(input("Old value: "))
    new = int(input("New value: "))
    
    for i in range(len(nums)):
        if nums[i] == old:
            nums[i] = new
    
    print("Updated list:", nums)

main()
```

    Enter numbers:  6 7 8 7 6 
    Old value:  1357
    New value:  2478
    

    Updated list: [6, 7, 8, 7, 6]
