# Program 36: Sort List
def main():
    nums = list(map(int, input("Enter numbers: ").split()))

    nums.sort()

    print("Sorted:", nums)

main()

Enter numbers:  67 89 7 8 8 9


Sorted: [7, 8, 8, 9, 67, 89]

# Program 37: Second Largest
def main():
    nums = list(map(int, input("Enter numbers: ").split()))

    nums = list(set(nums))
    nums.sort()

    print("Second largest:", nums[-2])

main()

Enter numbers:  6 78 9 8 5 4


Second largest: 9

# Program 38: Merge Lists
def main():
    a = list(map(int, input("Enter list1: ").split()))
    b = list(map(int, input("Enter list2: ").split()))

    merged = a + b

    print("Merged list:", merged)

main()

Enter list1:  4 6 7 8 6 5 2
Enter list2:  4 56 8 9 7 6


Merged list: [4, 6, 7, 8, 6, 5, 2, 4, 56, 8, 9, 7, 6]
