from datetime import datetime

def get_age(d):
    d1 = datetime.now()
    months = (d1.year - d.year) * 12 + d1.month - d.month

    year = int(months / 12)
    return year

age = get_age(datetime(1991, 1, 1))

def count_chars():
    text = input("Enter string: ")

    vowels = "aeiouAEIOU"

        if ch.isalpha():

                v += 1
            else:
                c += 1

    print("Vowels:", v)
