# Program 67: Star Pattern
def main():
    n = int(input("Enter rows: "))

    for i in range(1, n+1):
        for j in range(i):
            print("*", end="")
        print()

main()

# Program 68: Sum Even
def main():
    n = int(input("Enter limit: "))

    total = 0

    for i in range(1, n+1):
        if i % 2 == 0:
            total += i

    print("Sum of even:", total)

main()
