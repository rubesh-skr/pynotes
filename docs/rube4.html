##### Program 5: Largest number
def main():
    a = int(input("Enter first number: "))
    b = int(input("Enter second number: "))

    print("Comparing...")

    if a > b:
        print("Largest is:", a)
    else:
        print("Largest is:", b)

main()

Enter first number:  5
Enter second number:  8


Comparing...
Largest is: 8

# Program 6: Swap numbers
def main():
    a = 5
    b = 10

    print("Before swap:", a, b)

    temp = a
    a = b
    b = temp

    print("After swap:", a, b)

main()

Before swap: 5 10
After swap: 10 5
