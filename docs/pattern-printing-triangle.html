def sum_digits():
    num = int(input("Enter number: "))
    total = 0

    while num > 0:
        digit = num % 10
        total += digit
        num = num // 10

    print("Sum of digits:", total)

def repeat():
    n = int(input("How many numbers: "))
    for i in range(n):
        sum_digits()

def pattern():
    n = int(input("Enter rows: "))

    i = 1
    while i <= n:
        j = 1
        while j <= i:
            print("*", end=" ")
            j += 1
        print()
        i += 1

def repeat():
    t = int(input("How many patterns: "))
    for i in range(t):
        pattern()
