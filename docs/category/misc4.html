def main():
    num = int(input("Enter number: "))

    flag = 0

    for i in range(2, num):
        if num % i == 0:
            flag = 1

    if flag == 0 and num > 1:
        print("Prime number")
    else:
        print("Not prime")

main()

def main():
    num = int(input("Enter number: "))

    total = 0

    while num > 0:
        digit = num % 10
        total = total + digit
        num = num // 10

    print("Sum of digits:", total)

main()
