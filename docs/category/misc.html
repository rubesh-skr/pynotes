def linear_search():
    n = int(input("Enter number of elements: "))
    arr = []

    for i in range(n):
        val = int(input("Enter number: "))
        arr.append(val)

    key = int(input("Enter number to search: "))
    found = False

    for i in range(len(arr)):
        if arr[i] == key:
            print("Found at position", i + 1)
            found = True
            break

    if not found:
        print("Not found")

def bubble_sort():
    n = int(input("Enter count: "))
    arr = []

    for i in range(n):
        arr.append(int(input("Enter number: ")))

    i = 0
    while i < n:
        j = 0
        while j < n - i - 1:
            if arr[j] > arr[j + 1]:
                temp = arr[j]
                arr[j] = arr[j + 1]
                arr[j + 1] = temp
            j += 1
        i += 1

    print("Sorted list:", arr)

while True:

def count_chars():
    text = input("Enter string: ")

    vowels = "aeiouAEIOU"

        if ch.isalpha():
            if ch in vowels:
                v += 1
            else:
                c += 1

    print("Vowels:", v)
    print("Consonants:", c)
