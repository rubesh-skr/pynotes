# Program 22: Alphabet Check
def main():
    ch = input("Enter character: ")

    print("Checking...")

    if ch.isalpha():
        print("It is an alphabet")
    else:
        print("Not an alphabet")

main()

# Program 23: Count Vowels
def main():
    text = input("Enter string: ")

    count = 0

    for ch in text:
        if ch in "aeiouAEIOU":
            count += 1

    print("Vowel count:", count)

main()

Vowel count: 2
