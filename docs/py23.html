# Program 64: Count Characters
def main():
    text = input("Enter text: ")

    count = {}

    for ch in text:
        count[ch] = count.get(ch, 0) + 1

    print("Counts:", count)

main()

Enter text:  rubi


Counts: {'r': 1, 'u': 1, 'b': 1, 'i': 1}

# Program 65: Simple Interest
