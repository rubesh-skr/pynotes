# Program 39: Common Elements
def main():
    a = list(map(int, input("Enter list1: ").split()))
    b = list(map(int, input("Enter list2: ").split()))

    common = []

    for i in a:
        if i in b:
            common.append(i)

    print("Common:", common)

main()

Enter list1:  5 6 7 8 
Enter list2:  8 7 6 5 4


Common: [5, 6, 7, 8]

# Program 40: Frequency Count
def main():
    nums = list(map(int, input("Enter numbers: ").split()))

    freq = {}

    for n in nums:
        freq[n] = freq.get(n, 0) + 1

    print("Frequency:", freq)

main()

Enter numbers:  7 8 9 7 6 5 4


Frequency: {7: 2, 8: 1, 9: 1, 6: 1, 5: 1, 4: 1}

# Program 41: Create Dictionary
def main():
    keys = input("Enter keys: ").split()
    values = input("Enter values: ").split()

    data = {}

    for i in range(len(keys)):
        data[keys[i]] = values[i]

    print("Dictionary:", data)

main()

Enter keys:  6 7 8 9 5 4
Enter values:  6 7 8 9 5 4


Dictionary: {'6': '6', '7': '7', '8': '8', '9': '9', '5': '5', '4': '4'}
