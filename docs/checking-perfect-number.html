    print("Simple Interest:", si)

def reverse_number():
    num = int(input("Enter number: "))
    temp = num
    rev = 0

    while temp > 0:
        digit = temp % 10
        rev = rev * 10 + digit
        temp = temp // 10

    print("Original:", num)
    print("Reversed:", rev)

while True:

def perfect_number():
    num = int(input("Enter number: "))
    total = 0

    i = 1
    while i < num:
        if num % i == 0:
            total += i
        i += 1

    print("Sum of factors:", total)

    if total == num:
        print("Perfect number")
    else:
        print("Not a perfect number")
