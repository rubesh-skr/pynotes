# Program 10: Sum of N numbers
def main():
    n = int(input("Enter value: "))

    total = 0

    for i in range(1, n+1):
        total += i

    print("Sum is:", total)

main()

Enter value:  24


Sum is: 300

# Program 11: Multiplication Table
def main():
    num = int(input("Enter number: "))

    print("Multiplication Table:")

    for i in range(1, 11):
        result = num * i
        print(num, "x", i, "=", result)

main()
