# Program 87: Reverse Words
def main():
    text = input("Enter sentence: ")

    words = text.split()

    rev = words[::-1]

    print("Reversed words:", " ".join(rev))

main()

Enter sentence:  madam


Reversed words: madam

# Program 88: Capitalize
def main():
    text = input("Enter sentence: ")

    words = text.split()

    result = []

    for w in words:
        result.append(w.capitalize())

    print("Result:", " ".join(result))

main()

Enter sentence:  rb


Result: Rb

# Program 89: Substring
